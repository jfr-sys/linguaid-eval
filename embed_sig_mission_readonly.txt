#!/usr/bin/env python3
"""
embed_signature_mission.py

Overlays a trainer's typed signature image onto the confirmation de
mission PDF, at the {{SIGNATURE_PLACEHOLDER}} location (Linguaid's side
already carries the baked-in cachet from the docx template - this script
only ever touches the trainer's cell).

Usage: python3 embed_signature_mission.py '<json args>'
json args: {
  "pdfPath": "...unsigned pdf...",
  "signatureImg": "data:image/png;base64,...." or raw base64,
  "typedName": "Anna Malzy",
  "timestamp": "2026-07-20T12:00:00.000Z",
  "signerIp": "1.2.3.4",
  "outputPath": "...signed pdf..."
}
Prints {"success": true, "pdfPath": "..."} or {"success": false, "error": "..."}

Coordinates are calibrated against CONFIRMATION_MISSION_TEMPLATE.docx's
layout specifically (letter page, 612x792pt). If that template's layout
changes, SIG_X/SIG_Y below need recalibrating the same way: render the
filled docx to PDF, `pdftotext -bbox` the {{SIGNATURE_PLACEHOLDER}} token,
convert its top-left corner to bottom-origin points.
"""
import sys
import json
import base64
import io

def main():
    try:
        args = json.loads(sys.argv[1])
        pdf_path = args["pdfPath"]
        sig_data = args["signatureImg"]
        typed_name = args.get("typedName", "")
        timestamp = args.get("timestamp", "")
        signer_ip = args.get("signerIp", "")
        output_path = args["outputPath"]

        if sig_data.startswith("data:"):
            sig_data = sig_data.split(",", 1)[1]
        sig_bytes = base64.b64decode(sig_data)

        from reportlab.pdfgen import canvas
        from reportlab.lib.utils import ImageReader
        from pypdf import PdfReader, PdfWriter

        reader = PdfReader(pdf_path)
        num_pages = len(reader.pages)
        last_page = reader.pages[num_pages - 1]
        page_w = float(last_page.mediabox.width)
        page_h = float(last_page.mediabox.height)

        # Calibrated against the trainer's signature-cell token position
        # (see docstring). Values in PDF points, origin bottom-left.
        # Recalibrated 2026-07-22 against Joss's actual uploaded template
        # (different margins/spacing than the original build -> different Y).
        SIG_MAX_W = 130
        SIG_MAX_H = 45
        SIG_X = 289
        SIG_BASELINE_Y = 792 - 372.9  # bottom of the token's bbox

        img = ImageReader(io.BytesIO(sig_bytes))
        iw, ih = img.getSize()
        scale = min(SIG_MAX_W / iw, SIG_MAX_H / ih)
        draw_w, draw_h = iw * scale, ih * scale

        overlay_buf = io.BytesIO()
        c = canvas.Canvas(overlay_buf, pagesize=(page_w, page_h))
        c.drawImage(
            img, SIG_X, SIG_BASELINE_Y, width=draw_w, height=draw_h,
            mask="auto", preserveAspectRatio=True,
        )
        c.setFont("Helvetica", 7)
        c.setFillGray(0.4)
        meta = f"Signé électroniquement par {typed_name} — {timestamp} — IP {signer_ip}"
        c.drawString(SIG_X, SIG_BASELINE_Y - 10, meta[:110])
        c.save()
        overlay_buf.seek(0)

        overlay_reader = PdfReader(overlay_buf)
        overlay_page = overlay_reader.pages[0]

        writer = PdfWriter()
        for i, page in enumerate(reader.pages):
            if i == num_pages - 1:
                page.merge_page(overlay_page)
            writer.add_page(page)

        with open(output_path, "wb") as f:
            writer.write(f)

        print(json.dumps({"success": True, "pdfPath": output_path}))
    except Exception as e:
        print(json.dumps({"success": False, "error": str(e)}))
        sys.exit(1)

if __name__ == "__main__":
    main()
